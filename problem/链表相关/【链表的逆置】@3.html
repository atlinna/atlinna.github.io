    链表的逆置

    # 链表的逆置

    # 逆转完整列表

    var reverseList = function (head) {
    if (head == null || head.next == null) {
        return head;
    }
    var last = reverseList(head.next);
    head.next.next = head;
    head.next = null;
    return last;
};

    # 逆转前N个节点

    var successor = null;
var reverseN = function (head, n) {
    if (n == 1) {
        successor = head.next;
        return head;
    }
    var last = reverseN(head.next, n - 1);
    head.next.next = head;
    head.next = successor;
    return last;
}

    # 逆转区间内节点

    如果左边界为1，那么就相当于反转前right个节点，如果左边界不为1，那么通过递归，将找到left左边界等于1的节点。

    var reverseBetween = function (head, left, right) {
    if (left == 1) {
        return reverseN(head, right);
    }

    var last = reverseBetween(head.next, left - 1, right - 1);
    head.next = last
    return head;
};
