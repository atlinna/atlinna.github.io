    位运算 可换算为乘法。

    2 可即为 1 左移1位 4 可即为 1 左移2位 8 可即为 1 左移3位

    即：

    1 << 1 == 2;
1 << 2 == 4;
1 << 3 == 8;

    如何不使用 * | / | mod 来计算有符号除法，默认计算范围为【-(2 ** 31),2**31 - 1】

    const MAX_VAL = 2 ** 31 - 1, MIN_VAL = -(2 ** 31);
    if (dividend >= MAX_VAL) {
        if (divisor == 1) return MAX_VAL;
        if (divisor == -1) return -MAX_VAL;
    } else if (dividend == MIN_VAL) {
        if (divisor == 1) return MIN_VAL;
        if (divisor == -1) return MAX_VAL;
    }
    var flag = false;
    if (dividend > 0) {
        flag = !flag;
        dividend = -dividend;
    }
    if (divisor > 0) {
        flag = !flag;
        divisor = -divisor;
    }
    var stack = [divisor];
    var index = 0;
    while (true) {
        var sum = stack[index] + stack[index];
        if (sum >= dividend) {
            stack.push(sum)
        } else {
            break;
        }
        index += 1;
    }
    var side = 0;
    for (var i = stack.length - 1; i >= 0; i--) {
        if (stack[i] >= dividend) {
            side += 1 << i;
            dividend -= stack[i];
        }
    }
    return flag ? -side : side;
